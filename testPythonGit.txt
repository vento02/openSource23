import sys
k=[ list(map(int,sys.stdin.readline().split())) for j in range(9)]
max = k[0][0]
index = [1 , 1]
for i in range(9):
    for j in range(9):
        if max < k[i][j]:
            max=k[i][j]
            index[0]=i+1
            index[1]=j+1



print(str(index[0])+' '+str(index[1]))
